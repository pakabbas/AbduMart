Database: abdu_mart

TABLE: users
- id INT PK AUTO_INCREMENT
- email VARCHAR(255) UNIQUE
- password_hash VARCHAR(255)
- first_name VARCHAR(100)
- last_name VARCHAR(100)
- phone VARCHAR(20)
- role ENUM('customer','admin') DEFAULT 'customer'
- created_at TIMESTAMP

TABLE: categories
- id INT PK AUTO_INCREMENT
- clover_id VARCHAR(64) UNIQUE NULL
- name VARCHAR(255)
- sort_order INT DEFAULT 0
- image_url VARCHAR(500) NULL
- is_active TINYINT(1) DEFAULT 1
- synced_at TIMESTAMP NULL

TABLE: products
- id INT PK AUTO_INCREMENT
- clover_id VARCHAR(64) UNIQUE NULL
- category_id INT FK -> categories.id
- name VARCHAR(255)
- description TEXT NULL
- price DECIMAL(10,2)
- inventory INT DEFAULT 0
- image_url VARCHAR(500) NULL
- is_active TINYINT(1) DEFAULT 1
- synced_at TIMESTAMP NULL

TABLE: cart_items
- id INT PK AUTO_INCREMENT
- user_id INT FK -> users.id
- product_id INT FK -> products.id
- quantity INT DEFAULT 1
- UNIQUE(user_id, product_id)

TABLE: orders
- id INT PK AUTO_INCREMENT
- user_id INT FK -> users.id
- order_number VARCHAR(20) UNIQUE
- stripe_session_id VARCHAR(255) NULL
- stripe_payment_intent VARCHAR(255) NULL
- subtotal DECIMAL(10,2)
- tax DECIMAL(10,2) DEFAULT 0
- total DECIMAL(10,2)
- status ENUM('pending','paid','preparing','ready','picked_up','cancelled') DEFAULT 'pending'
- pickup_notes TEXT NULL
- customer_here_at TIMESTAMP NULL
- vehicle_description VARCHAR(255) NULL
- created_at TIMESTAMP
- updated_at TIMESTAMP

TABLE: order_items
- id INT PK AUTO_INCREMENT
- order_id INT FK -> orders.id
- product_id INT FK -> products.id
- product_name VARCHAR(255)
- quantity INT
- unit_price DECIMAL(10,2)
- line_total DECIMAL(10,2)

TABLE: clover_sync_log
- id INT PK AUTO_INCREMENT
- sync_type ENUM('full','categories','products')
- status ENUM('success','failed')
- message TEXT NULL
- created_at TIMESTAMP
